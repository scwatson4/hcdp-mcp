1) Full HCDP MCP Tool Design

Instead of mirroring the raw API 1:1, you should create semantic tools that map cleanly to how scientists think.

Below is a clean, production-ready MCP tool set.

🌧️ Raster & Grid Data Tools
1. get_raster_map

Purpose: Fetch a GeoTIFF raster

type get_raster_map = {
  date: string;                // "YYYY-MM" or ISO
  datatype: "rainfall" | "temperature" | "spi" | "ndvi_modis" | "ignition_probability";
  extent: "statewide" | "bi" | "ka" | "oa" | "mn";
  period: "month" | "day";
  production?: "new" | "legacy";
  aggregation?: "min" | "max" | "mean";
  timescale?: string;
};
2. get_raster_timeseries

Purpose: Time series at a point

type get_raster_timeseries = {
  start: string;
  end: string;
  lat?: number;
  lng?: number;
  row?: number;
  col?: number;
  datatype: string;
  extent: string;
  period: string;
  production?: string;
};
3. batch_raster_download

Purpose: Wrap /genzip

type batch_raster_download = {
  datasets: Array<{
    datatype: string;
    production?: string;
    aggregation?: string;
    timescale?: string;
    period: string;
    extent: string;
    start: string;
    end: string;
    files: ("data_map" | "metadata" | "anom" | "se")[];
  }>;
  delivery: "link" | "email" | "inline";
  email?: string;
};
